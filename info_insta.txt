<script>
const instaModal = document.getElementById('instaModal');
let instaTimer = null;

function showInsta(){
  instaModal.classList.add('visible');
  instaModal.setAttribute('aria-hidden','false');
  if(instaTimer) clearTimeout(instaTimer);
  instaTimer = setTimeout(hideInsta, 6000);
}
function hideInsta(){
  instaModal.classList.remove('visible');
  instaModal.setAttribute('aria-hidden','true');
  if(instaTimer){ clearTimeout(instaTimer); instaTimer = null; }
}

window.addEventListener('load', ()=> {
  setTimeout(showInsta, 1000);
});
</script>
